   function show_active() {
           var passwd = $.trim($('#inputPassword').val());
           if(!passwd)
           {
               alert('密码不能为空！');
               return false;
           }
            $.ajax({
                     async : true,
                     type:"get",
                     url:"https://api.studytime.xin/activationCode?passwd=" + passwd,
                     data:{},
                     dataType:"json",
                     success:function (e) {
                         if (e.status == 0) {
                            alert( e.message);
                            return false;
                         }else {
                            $('.jetbrains-code').append( "<td class=code><pre><span class=line>" +e.data+"</span><br></pre></td>");
                            $('.code_input').show();
                         }
                     },
                     error:function (e) {
                        alert( e.message);
                        return false;
                     }
            });
   }
</script>
